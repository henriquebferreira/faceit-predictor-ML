def get_mean_elo(data, team):
    mean_elos = []
    for team_data in data[team]:
        mean_team_elo = [p["elo"] for p in team_data]
        mean_elos.append(sum(mean_team_elo)/len(mean_team_elo))
    return mean_elos

batch_data["mean_elo_A"] = get_mean_elo(batch_data, 'teamA')
batch_data["mean_elo_B"] = get_mean_elo(batch_data, 'teamB')
batch_data["dif_mean_elo"] = batch_data["mean_elo_A"] - batch_data["mean_elo_B"]
In [33]: 
def get_winner(data):
    winners = []
    for s in data["score"]:
        team_rounds = [int(r) for r in s.split(" / ")]
        winner = 0 if team_rounds[0]>team_rounds[1]  else 1
        winners.append(winner)
    return winners

batch_data["winner"] = get_winner(batch_data)
In [38]: 
def get_is_upset(data):
    upsets = []
    for dif_mean_elo, winner in zip(data.dif_mean_elo, data.winner):
        if (dif_mean_elo >50 and winner ==1) or (dif_mean_elo<-50 and winner==0):
            upsets.append(1)
        else:
            upsets.append(0)
    return upsets

batch_data["is_upset"] = get_is_upset(batch_data)

CSGO is played under a "best-of" 30 rounds format, meaning that the first team to reach 16 rounds wins the match. Therefore, a minimum of 16 rounds should be played on every match.

However if after 30 rounds, the teams are tied 15-15, then an 6-round overtime is played to find the winner. Overtimes are repeated endlessly until a winner is found.

The data contains matches where the total number of rounds played is less than 16. Most likely it corresponds to forfeited matches .

Forfeited matches usually represent very unbalanced (one-sided) matches and must have atleast 11 played rounds.

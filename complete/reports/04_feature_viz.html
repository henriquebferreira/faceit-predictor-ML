Upsets vs Match ELO ¶

In [66]: 
UPSET_ELO_THRESHOLD = 75

def get_is_upset(data):
    upsets = []
    for dif_elo, winner in zip(data["dif_mean_elo"], data["winner"]):
        if dif_elo > UPSET_ELO_THRESHOLD and winner == 1 or dif_elo < -UPSET_ELO_THRESHOLD and winner == 0:
            upsets.append(1)
        else:
            upsets.append(0)
    return upsets

data["is_upset"] = get_is_upset(data)
